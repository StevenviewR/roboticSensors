import os
from collections import defaultdict

# Path to your YOLO labels folder
labels_folder = "/content/drive/My Drive/checking/export-dir0912_h/labels"

# Dictionary to store counts per class
class_counts = defaultdict(int)

# Loop through all .txt files in the labels folder
for filename in os.listdir(labels_folder):
    if filename.endswith(".txt"):
        file_path = os.path.join(labels_folder, filename)

        with open(file_path, "r") as f:
            lines = f.readlines()

        for line in lines:
            parts = line.strip().split()
            if not parts:
                continue
            try:
                class_id = int(parts[0])
                class_counts[class_id] += 1
            except ValueError:
                continue

# Print the results
print("Label counts per class ID:")
for class_id, count in sorted(class_counts.items()):
    print(f"Class {class_id}: {count}")

import os
import shutil
import random

# Paths
source_images = '/content/drive/My Drive/checking/export-dir0912_h/images'
source_labels = '/content/drive/My Drive/checking/export-dir0912_h/labels'
dest = '/content/drive/My Drive/checking/train20250912_h'
train_ratio = 0.8

# Get all image files
images = [f for f in os.listdir(source_images) if f.endswith(('.jpg', '.png', '.JPG'))]
random.shuffle(images)

# Split
train_size = int(len(images) * train_ratio)
train_images = images[:train_size]
val_images = images[train_size:]

# Helper function
def copy_files(file_list, split):
    os.makedirs(f'{dest}/images/{split}', exist_ok=True)
    os.makedirs(f'{dest}/labels/{split}', exist_ok=True)
    for img in file_list:
        base = os.path.splitext(img)[0]
        shutil.copy(f'{source_images}/{img}', f'{dest}/images/{split}/{img}')
        shutil.copy(f'{source_labels}/{base}.txt', f'{dest}/labels/{split}/{base}.txt')

copy_files(train_images, 'train')
copy_files(val_images, 'val')

import os
from collections import defaultdict

# Paths to your label directories
label_dirs = [
    "/content/drive/My Drive/checking/train20250912_h/labels/train",
    "/content/drive/My Drive/checking/train20250912_h/labels/val"
]

# Class names from your data.yaml
class_names = ['Overheating','Anomaly', 'Breakage', 'Below Normal']

# Dictionary to count instances per class
class_counts = defaultdict(int)

# Count instances from both train and val
for label_dir in label_dirs:
    if not os.path.exists(label_dir):
        print(f"⚠️ Skipping missing directory: {label_dir}")
        continue
    for filename in os.listdir(label_dir):
        if filename.endswith('.txt'):
            with open(os.path.join(label_dir, filename), 'r') as f:
                for line in f:
                    parts = line.strip().split()
                    if not parts:  # skip empty lines
                        continue
                    try:
                        class_id = int(parts[0])
                        if 0 <= class_id < len(class_names):
                            class_counts[class_id] += 1
                        else:
                            print(f"⚠️ Skipping invalid class ID {class_id} in {filename}")
                    except ValueError:
                        print(f"⚠️ Invalid line in {filename}: {line.strip()}")

# Print results
print("Class ID\tClass Name\t\tCount")
print("-" * 40)
for class_id in range(len(class_names)):
    name = class_names[class_id]
    count = class_counts[class_id]
    print(f"{class_id}\t\t{name:<20}\t{count}")

from ultralytics import YOLO
import glob
import shutil
import os

# This will automatically download yolov8n.pt if not found
model = YOLO('yolov8s.pt')

# Train the model
model.train(data="/content/drive/My Drive/checking/train20250912_h/data.yaml", epochs=500, imgsz=640)

# Validation performance
metrics = model.val()

# Find all weight files
weight_files = glob.glob('runs/detect/*/weights/*.pt')
print("Found weight files:", weight_files)

# Define destination folder in Google Drive
destination_folder = "/content/drive/MyDrive/Yolo_results/solar_faulty_detection20250912_h"
runs_dest_folder = os.path.join(destination_folder, "runs")

# Create folders if they don't exist
os.makedirs(destination_folder, exist_ok=True)
os.makedirs(runs_dest_folder, exist_ok=True)

print("Folders are ready at:", destination_folder)

# Copy each found weight file to Google Drive with renamed format
for file_path in weight_files:
    filename = file_path.split("/")[-1]                      # e.g., best.pt
    run_name = file_path.split("/")[2]                       # e.g., train233
    dest_path = os.path.join(destination_folder, f"{run_name}_{filename}")

    shutil.copy(file_path, dest_path)
    print(f"Copied: {file_path} → {dest_path}")

# Copy entire runs folder to destination_folder/runs
shutil.copytree("runs", runs_dest_folder, dirs_exist_ok=True)
print(f"Copied full 'runs' folder to: {runs_dest_folder}")
